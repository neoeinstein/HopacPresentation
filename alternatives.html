let CompareBool (comparand : ref<bool>)
                (input : Alt<'x>)
                (onTrue : 'x -> #Job<unit>)
                (onFalse : 'x -> #Job<unit>) : Job<unit> =
  input >>= fun x ->
    if !comparand then
      onTrue x :> Job<unit>
    else
      onFalse x :> Job<unit>

let Delay (duration : ref<TimeSpan>)
          (start : Alt<'x>)
          (stop : Alt<'y>)
          (onFinished : 'x -> #Job<unit>)
          (onAborted : 'y -> #Job<unit>) : Job<unit> =
  start >>= fun x ->
    Alt.choose
      [ stop                ^=> fun y -> onAborted y
        timeOut (!duration) ^=> fun () -> onFinished x ]
